    \section{What is integrability in elementary functions}
    \begin{frame}{What is integrability in elementary functions}
        \begin{definition}
            Field K is differential if it's equipped the unary function $'$ such that:
            \begin{itemize}
                \item $(a+b)' = a' + b'$
                \item $(ab)' = a'b + ab'$
            \end{itemize}
        \end{definition}
